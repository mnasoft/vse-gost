* Настройка PostgreSql под msys2
1. [ ] Установка
   #+begin_src sh
     pacman -S postgresql
   #+end_src
2. [ ] Инициализируем кластер базы данных
   #+begin_src shell
     initdb -D /var/lib/postgres/data --encoding=UTF8 --locale=ru_RU.UTF8 --lc-collate=ru_RU.UTF8 --lc-ctype=ru_RU.UTF8
     "D:\home\_namatv\PRG\msys64\ucrt64\bin\initdb.exe" "D:\home\_namatv\PRG\msys64\var\lib\postgres\data" --encoding=UTF8 --locale=ru_RU.UTF8 --lc-collate=ru_RU.UTF8 --lc-ctype=ru_RU.UTF8
   #+end_src

3. [ ] Запускаем сервер баз данных
   #+begin_src sh
     chcp 1251
     "D:\home\_namatv\PRG\msys64\ucrt64\bin\pg_ctl" -D "D:\home\_namatv\PRG\msys64\var\lib\postgres\data" -l journal.log start
   #+end_src

4. [ ] Останавливаем сервер баз данных
   #+begin_src sh
     chcp 1251     
     "D:\home\_namatv\PRG\msys64\ucrt64\bin\pg_ctl" -D "D:\home\_namatv\PRG\msys64\var\lib\postgres\data" -l journal.log stop   
   #+end_src
